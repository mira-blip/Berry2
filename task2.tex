\section{Operational Semantics}
Rule 1: \[\inferrule{x \in dom(\rhos) \quad \lambda^*\sqsubseteq\rhos(x)\quad\lambda^*\sqsubseteq\lambda }{\stmjudge [\lambda^*] {\texttt{fscanf}(\texttt{stdin}^{\lambda}, \dots, \& x)} {\rhos, chs[in^\lambda := v :: in^\lambda]} {\rhos \{x \mapsto v^{\lambda}\}, chs }}\]
Rule 2: \[\inferrule{x \in dom(\rhos)\quad \lambda^*\sqsubseteq\rhos(x)}{\stmjudge [\lambda^*] {\texttt{fscanf}(\texttt{stdin}^{\lambda}, \dots, \& x)} {\rhos, chs[in^\lambda := v :: in^\lambda]} {\rhos \{x \mapsto v^{\lambda \sqcup \lambda^*}\}, chs }}\]

\section{Information leaking code}

\lstset{style=ccode}

A program that leaks information if employing Rule 2:
\begin{lstlisting}
int main(){
    bool x, y;
    bool z = true;
    fscanf(stdin_hi, "%b", &x);
    fscanf(stdin_lo, "%b", &y);
    if(x) fscanf(stdin_lo, "%b", &y);
    if(y) z = false
    fprintf(stdout_lo, "%b", z)
}
\end{lstlisting}

\section{Derivations}

Derivation of \textbf{Rule 1} on \textbf{Test 1}:

\[
\begin{nd}
\der{
  \expjudge [L] \emptyenv {\lstinline!main ()!} {SecErr}
}{
  \subder{
    \stmjudge [L] {ss} {\emptyenv, chs} {SecErr}
  }{
    \subder{
      \stmjudge [L] {\lstinline!bool x, y!} {\emptyenv, \langle[\texttt{true}][][\texttt{true,false}][]\rangle} {\{x \mapsto ?^L, y \mapsto ?^L\}, chs}
      }{
      \axiom {x \notin dom(\rho)}
      \axiom {y \notin dom(\rho)}
    }
    \subder{
      \stmjudge[L]{\lstinline!bool z = true!}{\rhos, chs}{\rhos', chs}
    }{
      \axiom{z \notin dom(\rhos)}
      \axiom{\expjudge[p,L]{\rhos,chs}{\texttt{true}}{\texttt{true}^L}}
    }
    \subder{
      \stmjudge[L]{\texttt{fscanf}(\texttt{stdin}^{H},\dots, \&x)}{\{x \mapsto ?^L, y \mapsto ?^L\}, chs}{\{x \mapsto \texttt{true}^H, y \mapsto ?^L\}, \langle[],[],[\texttt{true,false}],[]\rangle}
    }{
      \axiom {x \in dom(\rhos)}
      \axiom {L \sqsubseteq (\rhos(x) = H)}
      \axiom {L \sqsubseteq H}
    }
    \subder{
      \stmjudge[L]{\texttt{fscanf}(\texttt{stdin}^{L},\dots, \&y)}{\{x \mapsto \texttt{true}^H, y \mapsto ?^L\}, chs'}{\{x \mapsto \texttt{true}^H, y \mapsto \texttt{true}^L\}, \langle[],[],[\texttt{false}],[]\rangle}
    }{
      \axiom {y \in dom(\rhos')}
      \axiom {L \sqsubseteq (\rhos'(y) = L)}
      \axiom {L \sqsubseteq L}
    }
    \subder{
      \stmjudge[L]{\utt{if}(x)\; \texttt{fscanf}(\texttt{stdin}^L, \dots, \&y)}{\rhos, chs''}{SecErr}
    }{
      \axiom{\expjudge[L]{\rhos', chs''}{x}{\texttt{true}^H, chs''}}
      \subder{
        \stmjudge[H]{\texttt{fscanf}(\texttt{stdin}^{L},\dots, \&y)}{\rhos', chs''}{SecErr}
      }{
        \axiom {y \in dom(\rhos')}
        \axiom {H \sqsubseteq (\rhos'(y) = \texttt{true}^H = H)}
        \axiom {H \not\sqsubseteq (\lambda = L)}
      }
    }
  }
}
\end{nd}
\]
